Detecting faces in the newspaper images¶

In [177]:
def detect_faces(image):
    '''Detects faces from given image
       @param image: A PIL.Image object.
       @param output: returns bounding boxes of detected faces.
    '''
    
    face_cascade = cv.CascadeClassifier("readonly/haarcascade_frontalface_default.xml")
    im = image.convert("RGB")
    im_cv = np.asarray(im)
    im_gray = cv.cvtColor(im_cv, cv.COLOR_BGR2GRAY)
    faces = face_cascade.detectMultiScale(im_gray, 1.69)
    bounding_box = []
    for x,y,w,h in faces:
        bounding_box.append((x, y, x+w, y+h))
    return bounding_box

Defining a face cropper from image¶

In [178]:
def face_crop(image, boundary):
    '''Crops detected faces from the image and creates a list of these image objects
        @param image: A PIL.Image object from which to crop the faces
        @param boundary: A list of tuples consisting coordinates of faces
        @param output: A list of PIL.Image objects of cropped faces
    '''
    faces = []
    im = image.copy()
    for box in boundary:
        face = im.crop(box)
        faces.append(face)
    return faces

Creating Contact sheets¶

In [182]:
for items in files:
    if "faces" in items.keys():
        first_image = items["faces"][0]
        width = first_image.width*4
        height = first_image.height*math.ceil(len(items["faces"])/4)
        contact_sheet = Image.new(first_image.mode, (width,height))
        x = 0
        y = 0
        for img in items["faces"]:
            contact_sheet.paste(img, (x, y))
            if x + first_image.width == contact_sheet.width:
                x = 0
                y = y + first_image.height
            else:
                x = x + first_image.width
        max_size = (600,600)
        contact_sheet.thumbnail(max_size)
        items.update({"contact_sheet": contact_sheet})
